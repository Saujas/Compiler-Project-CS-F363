
<<module square>>
takes input [arr: array[a..b] of integer];
start
	print(arr);
	arr[2]:= 100;
	print(arr);
end


<<<driver program>>>
start
	declare x, a, b:integer;
	a := 2;
	b := 4;
	declare arr: array[a..b] of integer;
	
	arr[2] := 2;
	arr[3] := 4;
	arr[4] := 6;

	use module square with parameters arr;

	print(arr);
end
